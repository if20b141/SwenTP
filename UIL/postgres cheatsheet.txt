CREATE TABLE IF NOT EXISTS Tours(id SERIAL PRIMARY KEY, tourname VARCHAR(50) UNIQUE NOT NULL, description VARCHAR(255) NOT NULL, startpoint VARCHAR(50) NOT NULL, endpoint VARCHAR(50) NOT NULL, type VARCHAR(50) NOT NULL, distance VARCHAR(50) NOT NULL, time VARCHAR(50) NOT NULL, information VARCHAR(100) NOT NULL);
CREATE TABLE IF NOT EXISTS Tourlogs(id SERIAL PRIMARY KEY, tourid INT NOT NULL, time VARCHAR(50) NOT NULL, distance VARCHAR(50) NOT NULL, rating VARCHAR(50) NOT NULL, comment VARCHAR(255));
